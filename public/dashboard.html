async function login() { const username = document.getElementById("username").value; const password = document.getElementById("password").value; const response = await fetch("/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ username, password }) }); const result = await response.json(); if (result.message) { // التحويل إلى لوحة التحكم بعد نجاح الدخول window.location.href = "/dashboard.html"; } else { alert(result.error); } }
